void setup(){
  pinMode(2,INPUT);
  //initiate the Mouse library
  Mouse.begin();
}

void loop(){
  //if the button is pressed, send a Right mouse click
  //then end the Mouse emulation
  if(digitalRead(2) == HIGH){
    Mouse.click();
    Mouse.end();
  }
}
